**Detailed HKDF-Extract Implementation** (Production-Ready)

Here is a complete, well-commented TypeScript implementation of the **HKDF-Extract** phase, following RFC 5869.

### 1. Core HKDF-Extract Function

```ts
import { createHmac } from 'crypto';

/**
 * HKDF Extract Phase (RFC 5869)
 * 
 * Turns input keying material (IKM) into a pseudorandom key (PRK)
 * using a salt.
 */
export function hkdfExtract(
  ikm: Buffer,                    // Master secret / Input Keying Material
  salt: Buffer = Buffer.alloc(0), // Optional salt
  hash: string = 'sha256'         // 'sha256' or 'sha512'
): Buffer {
  const hashLen = getHashLength(hash);

  // If salt is empty, use a string of zeros of hash length
  const zeroSalt = Buffer.alloc(hashLen, 0);

  return createHmac(hash, salt.length > 0 ? salt : zeroSalt)
    .update(ikm)
    .digest();
}

/** Helper: Get output length of the hash function */
function getHashLength(hash: string): number {
  switch (hash.toLowerCase()) {
    case 'sha256': return 32;
    case 'sha512': return 64;
    default:
      throw new Error(`Unsupported hash: ${hash}`);
  }
}
```

### 2. Full HKDF (Extract + Expand) – Recommended

```ts
/**
 * Full HKDF (Extract + Expand)
 */
export function hkdf(
  ikm: Buffer,
  salt: Buffer = Buffer.alloc(0),
  info: Buffer = Buffer.alloc(0),
  outputLength: number,
  hash: string = 'sha256'
): Buffer {
  const prk = hkdfExtract(ikm, salt, hash);
  return hkdfExpand(prk, info, outputLength, hash);
}

/**
 * HKDF Expand Phase
 */
export function hkdfExpand(
  prk: Buffer,
  info: Buffer = Buffer.alloc(0),
  outputLength: number,
  hash: string = 'sha256'
): Buffer {
  const hashLen = getHashLength(hash);
  let result = Buffer.alloc(0);
  let t = Buffer.alloc(0);
  let counter = 1;

  while (result.length < outputLength) {
    const hmac = createHmac(hash, prk);

    if (t.length > 0) hmac.update(t);
    hmac.update(info);
    hmac.update(Buffer.from([counter]));

    t = hmac.digest();
    result = Buffer.concat([result, t]);
    counter++;

    if (counter > 255) {
      throw new Error('HKDF expand: output too long');
    }
  }

  return result.subarray(0, outputLength);
}
```

### 3. TLS 1.3 Style HKDF-Expand-Label (Highly Recommended)

```ts
/**
 * HKDF-Expand-Label (exactly as used in TLS 1.3)
 */
export function hkdfExpandLabel(
  secret: Buffer,
  label: string,           // e.g. "kwp2000 security level 1"
  context: Buffer = Buffer.alloc(0),
  outputLength: number,
  hash: string = 'sha256'
): Buffer {
  const fullLabel = Buffer.concat([
    Buffer.from(`kwp2000 ${label}`),   // or "uds "
    context
  ]);

  return hkdfExpand(secret, fullLabel, outputLength, hash);
}
```

### 4. Usage Example in Security Access

```ts
const masterSecret = Buffer.from('your-master-secret-hex', 'hex');

function deriveSecurityKey(seed: Buffer, level: number): Buffer {
  return hkdfExpandLabel(
    masterSecret,
    `security-access-level-${level}`,
    seed,                    // ECU-generated seed as context
    16,                      // 16 bytes = AES-128 key
    'sha256'
  );
}
```

### Best Practices

- Use **SHA-256** for most cases (excellent security/performance balance).
- Always include meaningful `info` / label for domain separation.
- Use a fixed but unique `salt` per ECU family or OEM.
- Validate output length (max 255 × hashLen).
