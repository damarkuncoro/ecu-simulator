### 1. Simple XOR Algorithm (Most Common)

```ts
class XorSecurityAlgorithm {
  private mask: Buffer;

  constructor(mask: Buffer | number[]) {
    this.mask = Buffer.from(mask);
  }

  calculateKey(seed: Buffer, level: number): Buffer {
    const key = Buffer.alloc(seed.length);
    for (let i = 0; i < seed.length; i++) {
      key[i] = seed[i] ^ this.mask[i % this.mask.length];
    }
    return key;
  }
}

// Usage
const algo = new XorSecurityAlgorithm([0x55, 0xAA, 0x33, 0xCC]); // 4-byte mask example
const seed = Buffer.from([0x12, 0x34, 0x56, 0x78]);
const key = algo.calculateKey(seed, 1);
console.log(key.toString('hex')); // e.g. "47 9e 65 b4"
```

### 2. Linear Congruential (Multiplicative) Algorithm

```ts
class LinearSecurityAlgorithm {
  private multiplier: bigint;
  private adder: bigint;
  private modulus: bigint;

  constructor(multiplier: number, adder: number, bits: number = 32) {
    this.multiplier = BigInt(multiplier);
    this.adder = BigInt(adder);
    this.modulus = BigInt(1) << BigInt(bits);
  }

  calculateKey(seedBuf: Buffer): Buffer {
    let seed = BigInt('0x' + seedBuf.toString('hex'));
    const result = (seed * this.multiplier + this.adder) % this.modulus;
    
    // Convert back to same byte length
    const hex = result.toString(16).padStart(seedBuf.length * 2, '0');
    return Buffer.from(hex, 'hex');
  }
}

// Example: Common 16-bit style
const algo = new LinearSecurityAlgorithm(0x1234, 0x5678, 16);
```

### 3. Rotation + XOR (Seen in Some ECUs)

```ts
function rotateLeft(value: number, bits: number, size: number): number {
  return ((value << bits) | (value >> (size - bits))) & ((1 << size) - 1);
}

class RotateXorAlgorithm {
  calculateKey(seed: Buffer): Buffer {
    const key = Buffer.alloc(seed.length);
    
    for (let i = 0; i < seed.length; i++) {
      let val = seed[i];
      val = rotateLeft(val, 3, 8);        // Rotate left 3 bits
      val = val ^ 0xA5;                   // XOR with constant
      val = (val + i) & 0xFF;             // Add index for variation
      key[i] = val;
    }
    return key;
  }
}
```

### 4. Example from Real ECU (SIMK43-style 16-bit)

```ts
function calculateSIMKKey(seed: number): number {  // seed as 16-bit int
  let key = 0x9360;
  for (let i = 0; i < 0x24; i++) {  // 36 iterations
    key = ((key * 2) ^ seed) & 0xFFFF;
  }
  return key;
}

// Buffer version
function calculateSIMKKeyBuffer(seedBuf: Buffer): Buffer {
  if (seedBuf.length !== 2) throw new Error("Expected 2-byte seed");
  const seed = seedBuf.readUInt16BE(0);
  const key = calculateSIMKKey(seed);
  const keyBuf = Buffer.alloc(2);
  keyBuf.writeUInt16BE(key);
  return keyBuf;
}
```

### 5. Plugin-Friendly Registry (Recommended for Your Project)

```ts
interface SecurityAlgorithm {
  calculate(seed: Buffer, level: number, context?: any): Buffer;
}

class SecurityEngine {
  private algorithms = new Map<string, SecurityAlgorithm>();

  register(oem: string, level: number, algo: SecurityAlgorithm) {
    const key = `${oem.toLowerCase()}:${level}`;
    this.algorithms.set(key, algo);
  }

  getKey(oem: string, level: number, seed: Buffer): Buffer {
    const algo = this.algorithms.get(`${oem.toLowerCase()}:${level}`);
    if (!algo) {
      throw new Error(`No algorithm registered for ${oem} level ${level}`);
    }
    return algo.calculate(seed, level);
  }
}

// Usage in your ECU
const security = new SecurityEngine();
security.register('bosch', 1, new XorSecurityAlgorithm([0x55, 0xAA, 0x33, 0xCC]));
security.register('vag', 1, new LinearSecurityAlgorithm(0xA5A5, 0x5A5A));
```
