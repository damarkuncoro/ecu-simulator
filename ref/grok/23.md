### 1. Full HKDF Implementation with SHA-512 Support

```ts
import { createHmac } from 'crypto';

/** Get hash output length */
function getHashLength(hash: string): number {
  switch (hash.toLowerCase()) {
    case 'sha256': return 32;
    case 'sha512': return 64;
    default:
      throw new Error(`Unsupported hash: ${hash}`);
  }
}

/**
 * HKDF Extract (SHA-512 supported)
 */
export function hkdfExtract(
  ikm: Buffer,
  salt: Buffer = Buffer.alloc(0),
  hash: string = 'sha256'
): Buffer {
  const hashLen = getHashLength(hash);
  const zeroSalt = Buffer.alloc(hashLen, 0);

  return createHmac(hash, salt.length > 0 ? salt : zeroSalt)
    .update(ikm)
    .digest();
}

/**
 * HKDF Expand
 */
export function hkdfExpand(
  prk: Buffer,
  info: Buffer = Buffer.alloc(0),
  outputLength: number,
  hash: string = 'sha256'
): Buffer {
  const hashLen = getHashLength(hash);
  let result = Buffer.alloc(0);
  let t = Buffer.alloc(0);
  let counter = 1;

  while (result.length < outputLength) {
    const hmac = createHmac(hash, prk);

    if (t.length > 0) hmac.update(t);
    hmac.update(info);
    hmac.update(Buffer.from([counter]));

    t = hmac.digest();
    result = Buffer.concat([result, t]);
    counter++;

    if (counter > 255) throw new Error('HKDF output too long');
  }

  return result.subarray(0, outputLength);
}

/**
 * Full HKDF (Extract + Expand)
 */
export function hkdf(
  ikm: Buffer,
  salt: Buffer = Buffer.alloc(0),
  info: Buffer = Buffer.alloc(0),
  outputLength: number,
  hash: string = 'sha256'
): Buffer {
  const prk = hkdfExtract(ikm, salt, hash);
  return hkdfExpand(prk, info, outputLength, hash);
}

/**
 * TLS-style HKDF-Expand-Label
 */
export function hkdfExpandLabel(
  secret: Buffer,
  label: string,
  context: Buffer = Buffer.alloc(0),
  outputLength: number,
  hash: string = 'sha256'
): Buffer {
  const fullInfo = Buffer.concat([
    Buffer.from(`kwp2000 ${label}`),
    context
  ]);
  return hkdfExpand(secret, fullInfo, outputLength, hash);
}
```

### 3. Usage Example

```ts
const master = Buffer.from('your-very-long-master-secret-key-here-512-bit-security', 'utf8');
const seed = Buffer.from('random-ecu-seed-12345678', 'utf8');

const key = hkdfExpandLabel(
  master,
  'security-access-level-1',
  seed,
  32,           // 32 bytes = AES-256 key
  'sha512'      // Use SHA-512
);

console.log('Derived Key:', key.toString('hex'));
```

**Recommendation**:  
Use **SHA-256** for most cases (faster, sufficient security).  
Use **SHA-512** when you need higher security margin or larger keys (e.g., AES-256 + MAC).
