### Quick Test Code (TypeScript)

```ts
import { hkdf } from './your-hkdf';

const ikm = Buffer.from('0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b', 'hex');

const output = hkdf(ikm, salt, info, 42);
```
